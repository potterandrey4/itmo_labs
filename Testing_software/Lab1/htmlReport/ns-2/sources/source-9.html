 package org.example.domain;
 
 import java.util.ArrayList;
 import java.util.List;
 
 /**
  * Главный класс, представляющий космическую симуляцию
  */
 public class SpaceSimulation {
     private final Screen screen;
     private final List<CelestialObject> celestialObjects;
     private SimulationState currentState;
 
     public SpaceSimulation() {
         this.screen = new Screen();
         this.celestialObjects = new ArrayList<>();
         this.currentState = SimulationState.NOT_STARTED;
     }
 
     /**
      * Запускает симуляцию
      */
     public void start() {
         this.currentState = SimulationState.STARTED;
         // Изначально на экране ничего не отображается
         screen.clearDisplay();
     }
 
     /**
      * Добавляет звезду в симуляцию
      * 
      * @param star Звезда для добавления
      */
     public void addStar(Star star) {
         celestialObjects.add(star);
         if (currentState == SimulationState.STARTED) {
             screen.displayObject(star);
         }
     }
 
     /**
      * Добавляет планету в симуляцию
      * 
      * @param planet Планета для добавления
      */
     public void addPlanet(Planet planet) {
         celestialObjects.add(planet);
         if (currentState == SimulationState.STARTED) {
             screen.displayObject(planet);
         }
     }
 
     /**
      * @return Текущее состояние симуляции
      */
     public SimulationState getCurrentState() {
         return currentState;
     }
 
     /**
      * @return Список всех космических объектов в симуляции
      */
     public List<CelestialObject> getCelestialObjects() {
         return new ArrayList<>(celestialObjects);
     }
 
     /**
      * @return Экран симуляции
      */
     public Screen getScreen() {
         return screen;
     }
 }
