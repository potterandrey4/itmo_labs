 package org.example.domain;
 
 import java.util.ArrayList;
 import java.util.List;
 
 /**
  * Класс, представляющий экран, на котором отображается симуляция
  */
 public class Screen {
     private final List<CelestialObject> displayedObjects;
     private boolean isEmpty;
 
     public Screen() {
         this.displayedObjects = new ArrayList<>();
         this.isEmpty = true;
     }
 
     /**
      * Очищает экран от всех объектов
      */
     public void clearDisplay() {
         displayedObjects.clear();
         isEmpty = true;
     }
 
     /**
      * Отображает космический объект на экране
      * 
      * @param object Объект для отображения
      */
     public void displayObject(CelestialObject object) {
         displayedObjects.add(object);
         isEmpty = false;
     }
 
     /**
      * @return Список всех отображаемых объектов
      */
     public List<CelestialObject> getDisplayedObjects() {
         return new ArrayList<>(displayedObjects);
     }
 
     /**
      * @return true, если на экране ничего не отображается
      */
     public boolean isEmpty() {
         return isEmpty;
     }
 }
